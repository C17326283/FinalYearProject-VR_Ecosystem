            tree("GetWanderTarget")

    sequence
        AvoidPredators()
        MoveInDirection()
        sequence
            CheckGettingCloser()

        fallback
            sequence
                IsHungry()
                fallback
                    sequence
                        GetAnimalTarget()
                        ChasePrey()
                    sequence
                        CheckIfRemembersResource("Food")
                        SeekTarget()
            sequence
                IsThirsty()
                CheckIfRemembersResource("Water")
                SeekTarget()
        sequence
            CheckArrivedAtTarget()
            fallback
                AttackTarget()
                ConsumeTarget()
