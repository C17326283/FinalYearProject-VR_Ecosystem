tree("Root")
    fallback 
        tree("CheckForEnemies")
        tree("Hunt")
        tree("Wander")
	    
tree("CheckForEnemies")
    sequence
        CheckIfEnemyClose()
        FleeFromEnemy()
            
tree("Hunt")
    sequence
        IsHungry()
        CheckIfFoodClose()
        fallback
            sequence
                SeekTarget()
                CheckGettingCloser()
            sequence
                ArriveTarget()
                fallback
                    AttackTarget()
                    EatTarget()
           
tree("Wander")
    sequence
        fallback
            CheckWanderTarget()
            GetWanderTarget()
        fallback
            sequence
                SeekTarget()
                CheckGettingCloser()
            GetWanderTarget()
